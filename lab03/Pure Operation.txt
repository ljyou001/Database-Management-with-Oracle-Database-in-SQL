
SELECT SSN,HOURS
FROM WORKS_ON
WHERE PNUMBER = 30 AND HOURS>=15 AND HOURS<=30;


SELECT SSN, NAME, SALARY, SUPER_SSN, DNUMBER
FROM EMPLOYEES E
WHERE DNUMBER = 4
ORDER BY SALARY DESC;


SELECT E1.NAME
FROM EMPLOYEES E1, EMPLOYEES E2
WHERE E1.SUPER_SSN = E2.SSN AND E2.SUPER_SSN IS NULL;


SELECT NAME
FROM EMPLOYEES
WHERE NAME LIKE '%a%';


SELECT SSN
FROM WORKS_ON W, PROJECTS P
WHERE W.PNUMBER = P.PNUMBER AND P.PNAME = 'Reorganization'
MINUS
SELECT SSN
FROM WORKS_ON W, PROJECTS P
WHERE W.PNUMBER = P.PNUMBER AND (P.PNAME = 'ProductZ' OR P.PNAME = 'Newbenefits');


SELECT DISTINCT NAME 
FROM EMPLOYEES E, WORKS_ON W1, WORKS_ON W2
WHERE E.SSN = W1.SSN AND E.SSN = W2.SSN AND W1.PNUMBER <> W2.PNUMBER;


SELECT DISTINCT E1.NAME, E1.SALARY
FROM EMPLOYEES E1, EMPLOYEES E2
WHERE E1.SALARY < E2.SALARY;


SELECT NAME,SALARY
FROM EMPLOYEES
MINUS
SELECT DISTINCT E1.NAME, E1.SALARY
FROM EMPLOYEES E1, EMPLOYEES E2
WHERE E1.SALARY < E2.SALARY;
